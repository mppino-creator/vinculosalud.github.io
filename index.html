// ============================================================ // main.js - VERSIÓN SIMPLIFICADA PARA DIRECTORIO DE PSICÓLOGOS // Solo carga profesionales desde Firebase y muestra contactos. // Sin reservas, sin administración, sin paneles. // ============================================================ // Importar configuración de Firebase (debe existir en js/config/firebase.js) import { db } from './config/firebase.js'; // ============================================================ // 1. FUNCIONES DE UTILIDAD // ============================================================ function showToast(mensaje, tipo = 'info') { // No se usa en esta versión, pero se deja por compatibilidad console.log(`[${tipo}] ${mensaje}`); } function formatRut(input) { // No se usa, pero se deja por si acaso let valor = input.value.replace(/\./g, '').replace(/\-/g, ''); if (valor.length <= 1) return; let cuerpo = valor.slice(0, -1); let dv = valor.slice(-1); cuerpo = cuerpo.replace(/\B(?=(\d{3})+(?!\d))/g, "."); input.value = cuerpo + "-" + dv; } // ============================================================ // 2. CARGAR PROFESIONALES DESDE FIREBASE // ============================================================ let todosLosProfesionales = []; function cargarProfesionales() { const ref = db.ref('staff'); ref.on('value', (snapshot) => { const data = snapshot.val(); const grid = document.getElementById('psychGrid'); if (!data) { grid.innerHTML = `
No hay profesionales registrados aún.
`; return; } todosLosProfesionales = Object.keys(data).map(key => ({ id: key, ...data[key] })); poblarFiltroEspecialidades(todosLosProfesionales); renderizarProfesionales(todosLosProfesionales); }, (error) => { console.error('Error al cargar staff:', error); document.getElementById('psychGrid').innerHTML = `
Error al cargar los profesionales. Verifica tu conexión o configuración de Firebase.
`; }); } // ============================================================ // 3. RENDERIZAR TARJETAS DE PROFESIONALES // ============================================================ function renderizarProfesionales(lista) { const grid = document.getElementById('psychGrid'); if (!lista || lista.length === 0) { grid.innerHTML = `
No hay profesionales que coincidan con tu búsqueda.
`; return; } let html = ''; lista.forEach(psych => { const nombre = psych.name || 'Sin nombre'; const foto = psych.photo || 'https://images.unsplash.com/photo-1559839734-2b71ea197ec2?w=400&h=400&fit=crop'; const especialidades = Array.isArray(psych.specialties) ? psych.specialties.join(', ') : ''; const bio = psych.bio || ''; const whatsapp = psych.whatsapp || psych.phone || ''; const instagram = psych.instagram || ''; const telefono = psych.phone || ''; const direccion = psych.address || ''; let waNumero = whatsapp.replace(/[^0-9+]/g, ''); if (waNumero && !waNumero.startsWith('+')) waNumero = '+' + waNumero; html += `

${nombre}

${especialidades || 'Sin especialidades'}
${bio ? `
${bio}
` : ''}
${instagram ? `` : ''} ${waNumero ? `` : ''} ${telefono ? ` ${telefono}` : ''} ${direccion ? ` ${direccion}` : ''}
${waNumero ? ` Contactar ` : (telefono ? ` Llamar ` : '')}
`; }); grid.innerHTML = html; } // ============================================================ // 4. FILTRO DE ESPECIALIDADES // ============================================================ function poblarFiltroEspecialidades(profesionales) { const select = document.getElementById('specialtyFilter'); if (!select) return; const especialidades = new Set(); profesionales.forEach(p => { if (Array.isArray(p.specialties)) { p.specialties.forEach(s => especialidades.add(s)); } }); let options = ''; especialidades.forEach(esp => { options += ``; }); select.innerHTML = options; } // ============================================================ // 5. FILTRAR PROFESIONALES (desde la interfaz) // ============================================================ function filtrarProfesionales() { const search = document.getElementById('searchFilter')?.value?.toLowerCase() || ''; const specialty = document.getElementById('specialtyFilter')?.value || ''; let filtrados = todosLosProfesionales.filter(p => { const nombre = (p.name || '').toLowerCase(); const bio = (p.bio || '').toLowerCase(); const specs = Array.isArray(p.specialties) ? p.specialties.join(' ').toLowerCase() : ''; const matchSearch = nombre.includes(search) || bio.includes(search) || specs.includes(search); const matchSpec = specialty === '' || (Array.isArray(p.specialties) && p.specialties.includes(specialty)); return matchSearch && matchSpec; }); renderizarProfesionales(filtrados); } // ============================================================ // 6. NAVEGACIÓN SIMPLE // ============================================================ function mostrarSeccion(id) { const secciones = ['inicio', 'about', 'equipo', 'contacto']; secciones.forEach(sec => { const el = document.getElementById(sec); if (el) { el.style.display = (sec === id) ? 'block' : 'none'; } }); const filters = document.getElementById('filtersContainer'); if (filters) { filters.style.display = (id === 'equipo') ? 'flex' : 'none'; } if (id === 'inicio') { window.scrollTo({ top: 0, behavior: 'smooth' }); } else { const target = document.getElementById(id); if (target) target.scrollIntoView({ behavior: 'smooth' }); } document.querySelectorAll('.nav-links a').forEach(link => { const href = link.getAttribute('href'); if (href === `#${id}`) { link.style.fontWeight = '700'; } else { link.style.fontWeight = '500'; } }); } // ============================================================ // 7. EXPONER FUNCIONES GLOBALES // ============================================================ window.showToast = showToast; window.formatRut = formatRut; window.mostrarSeccion = mostrarSeccion; window.filtrarProfesionales = filtrarProfesionales; window.cargarProfesionales = cargarProfesionales; // ============================================================ // 8. INICIALIZAR AL CARGAR LA PÁGINA // ============================================================ document.addEventListener('DOMContentLoaded', () => { if (db && db.ref) { cargarProfesionales(); } else { console.error('❌ Firebase no está inicializado correctamente.'); document.getElementById('psychGrid').innerHTML = `
Error de configuración de Firebase. Verifica que tu archivo config/firebase.js esté correcto.
`; } }); console.log('✅ main.js simplificado cargado (directorio de psicólogos)');
